void test(void)
{
	for (int j = 0; j < 100; j++) 
	{ 
		for (i = 100 - 1; i > j; i--) 
		{ 
		if (list[j] < list[i]) 
			{ 
			temp = list[j]; 
			list[j] = list[i]; 
			list[i] = temp; 
			} 
		}
	}
}
